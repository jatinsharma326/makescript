import { AbsoluteFill, useCurrentFrame, useVideoConfig, spring, interpolate } from "remotion";

const safeInterpolate = (value, inputRange, outputRange, options) => {
  if (!inputRange || !outputRange || inputRange.length < 2 || outputRange.length < 2) {
    return outputRange && outputRange[0] !== undefined ? outputRange[0] : 0;
  }
  const sortedPairs = inputRange.map((v, i) => ({ input: v, output: outputRange[i] }))
    .sort((a, b) => a.input - b.input);
  const uniquePairs = [];
  for (let i = 0; i < sortedPairs.length; i++) {
    if (i === sortedPairs.length - 1 || sortedPairs[i].input !== sortedPairs[i + 1].input) {
      uniquePairs.push(sortedPairs[i]);
    }
  }
  if (uniquePairs.length < 2) return outputRange[0] || 0;
  return interpolate(
    value,
    uniquePairs.map(p => p.input),
    uniquePairs.map(p => p.output),
    options || { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );
};


export const FocusMode = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();
  const localFrame = frame % durationInFrames;

  // Central coordinates for explosion
  const cx = 540;
  const cy = 960;

  // Mushroom cloud parameters
  const cloudGrowth = spring({
    frame: localFrame,
    fps,
    config: { mass: 1.2, stiffness: 80, damping: 25 },
    durationInFrames: 100,
  });
  const stemHeight = safeInterpolate(localFrame, [0, 60, 120], [0, 300, 600], {
    extrapolateRight: "clamp",
  });
  const capRadius = cloudGrowth * 180;
  const stemWidth = 60;

  // Flash parameters
  const flashOpacity = safeInterpolate(localFrame, [0, 10, 30], [1, 0.8, 0], {
    extrapolateRight: "clamp",
  });
  const flashRadius = safeInterpolate(localFrame, [0, 30], [80, 200], {
    extrapolateRight: "clamp",
  });

  // Shockwave rings
  const rings = [
    { delay: 0, maxR: 600, color: "#ff6600", strokeW: 4 },
    { delay: 8, maxR: 500, color: "#ff4400", strokeW: 3 },
    { delay: 16, maxR: 400, color: "#ff2200", strokeW: 2 },
  ];

  // Debris particles: 50 small circles flying outward
  const particleCount = 50;
  const debris = [];
  for (let i = 0; i < particleCount; i++) {
    const angle = (i / particleCount) * Math.PI * 2 + localFrame * 0.02;
    const speed = 2 + (i % 5) * 0.5;
    const dist = localFrame * speed * 1.2;
    const size = 2 + (i % 3);
    const color = i % 2 === 0 ? "#ff6600" : "#ffaa00";
    debris.push({ angle, speed, dist, size, color });
  }

  // Background particles (ash, dust)
  const bgParticles = 80;
  const bgPart = [];
  for (let i = 0; i < bgParticles; i++) {
    const x = (i * 137 + 50) % 1080;
    const y = (i * 97 + 30) % 1920;
    const driftX = Math.sin(i * 1.3 + localFrame * 0.02) * 30;
    const driftY = Math.sin(i * 2.1 + localFrame * 0.015) * 40;
    bgPart.push({ x, y, driftX, driftY, r: 1 + (i % 3), opacity: 0.2 + Math.sin(i + localFrame * 0.03) * 0.15 });
  }

  return (
    <AbsoluteFill
      style={{
        backgroundColor: "transparent",
      }}
    >
      <svg
        viewBox="0 0 1080 1920"
        style={{ width: "100%", height: "100%", position: "absolute" }}
      >
        {/* Background floating ash particles */}
        {bgPart.map((p, idx) => (
          <circle
            key={`bg-${idx}`}
            cx={p.x + p.driftX}
            cy={p.y + p.driftY}
            r={p.r}
            fill="#aaaaaa"
            opacity={p.opacity}
          />
        ))}

        {/* Mushroom cloud – stem */}
        <rect
          x={cx - stemWidth / 2}
          y={cy - stemHeight}
          width={stemWidth}
          height={stemHeight}
          rx={20}
          fill="#8b4513"
          opacity={0.8 + 0.2 * Math.sin(localFrame * 0.1)}
        />

        {/* Mushroom cloud – cap (drawn as multiple overlapping ellipses for a puffy look) */}
        <ellipse
          cx={cx}
          cy={cy - stemHeight - capRadius * 0.4}
          rx={capRadius}
          ry={capRadius * 0.6}
          fill="#ff4500"
          opacity={0.7}
        />
        <ellipse
          cx={cx - capRadius * 0.25}
          cy={cy - stemHeight - capRadius * 0.5}
          rx={capRadius * 0.5}
          ry={capRadius * 0.45}
          fill="#ff6600"
          opacity={0.6}
        />
        <ellipse
          cx={cx + capRadius * 0.2}
          cy={cy - stemHeight - capRadius * 0.55}
          rx={capRadius * 0.5}
          ry={capRadius * 0.5}
          fill="#ffaa00"
          opacity={0.5}
        />

        {/* Central flash */}
        <circle
          cx={cx}
          cy={cy}
          r={flashRadius}
          fill="#ffcc00"
          opacity={flashOpacity}
        />

        {/* Shockwave rings */}
        {rings.map((ring, idx) => {
          const r = safeInterpolate(
            localFrame,
            [0, 30, 60],
            [0, ring.maxR, ring.maxR + 100],
            { extrapolateRight: "clamp", extrapolateLeft: "clamp" }
          );
          const op = safeInterpolate(
            localFrame,
            [0, 40],
            [1, 0],
            { extrapolateRight: "clamp" }
          );
          return (
            <circle
              key={`ring-${idx}`}
              cx={cx}
              cy={cy}
              r={r + ring.delay * 2}
              fill="none"
              stroke={ring.color}
              strokeWidth={ring.strokeW}
              opacity={op * 0.8}
            />
          );
        })}

        {/* Debris particles */}
        {debris.map((p, idx) => {
          const dx = Math.cos(p.angle) * p.dist;
          const dy = Math.sin(p.angle) * p.dist - localFrame * 4; // upwards motion
          const opacity = Math.max(0, 1 - p.dist / 800);
          return (
            <circle
              key={`debris-${idx}`}
              cx={cx + dx}
              cy={cy + dy}
              r={p.size}
              fill={p.color}
              opacity={opacity}
            />
          );
        })}

        {/* Additional debris – random sparks */}
        {Array.from({ length: 20 }).map((_, i) => {
          const angle = (i / 20) * Math.PI * 2 + localFrame * 0.05;
          const dist = localFrame * (4 + (i % 3) * 1.5);
          return (
            <line
              key={`spark-${i}`}
              x1={cx + Math.cos(angle) * dist}
              y1={cy + Math.sin(angle) * dist}
              x2={cx + Math.cos(angle) * (dist + 6)}
              y2={cy + Math.sin(angle) * (dist + 6)}
              stroke={
                i % 2 === 0 ? "#ff7700" : "#ffcc00"
              }
              strokeWidth={1.5 + (i % 2)}
              opacity={Math.max(0, 1 - dist / 500)}
            />
          );
        })}
      </svg>
    </AbsoluteFill>
  );
};

export default FocusMode;
