import { AbsoluteFill, useCurrentFrame, useVideoConfig, spring, interpolate } from "remotion";

const safeInterpolate = (value, inputRange, outputRange, options) => {
  if (!inputRange || !outputRange || inputRange.length < 2 || outputRange.length < 2) {
    return outputRange && outputRange[0] !== undefined ? outputRange[0] : 0;
  }
  const sortedPairs = inputRange.map((v, i) => ({ input: v, output: outputRange[i] }))
    .sort((a, b) => a.input - b.input);
  const uniquePairs = [];
  for (let i = 0; i < sortedPairs.length; i++) {
    if (i === sortedPairs.length - 1 || sortedPairs[i].input !== sortedPairs[i + 1].input) {
      uniquePairs.push(sortedPairs[i]);
    }
  }
  if (uniquePairs.length < 2) return outputRange[0] || 0;
  return interpolate(
    value,
    uniquePairs.map(p => p.input),
    uniquePairs.map(p => p.output),
    options || { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );
};


export const FocusMode = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();
  const localFrame = frame;
  const totalFrames = 18 * 30; // 540

  // Flash size – fades out
  const flashRadius = safeInterpolate(localFrame, [0, 10, 30], [80, 120, 0], {
    extrapolateRight: "clamp",
  });
  const flashOpacity = safeInterpolate(localFrame, [0, 10, 30], [0.9, 0.6, 0], {
    extrapolateRight: "clamp",
  });

  // Shockwave ring sizes and opacity
  const ring1R = safeInterpolate(localFrame, [0, 40], [40, 400], { extrapolateRight: "clamp" });
  const ring1Opacity = safeInterpolate(localFrame, [0, 30, 40], [0.8, 0.4, 0], { extrapolateRight: "clamp" });
  const ring2R = safeInterpolate(localFrame, [5, 45], [30, 500], { extrapolateRight: "clamp" });
  const ring2Opacity = safeInterpolate(localFrame, [5, 35, 45], [0.7, 0.3, 0], { extrapolateRight: "clamp" });
  const ring3R = safeInterpolate(localFrame, [10, 50], [20, 600], { extrapolateRight: "clamp" });
  const ring3Opacity = safeInterpolate(localFrame, [10, 40, 50], [0.6, 0.2, 0], { extrapolateRight: "clamp" });

  // Mushroom cloud growth
  const stemHeight = safeInterpolate(localFrame, [5, 40], [0, 250], { extrapolateRight: "clamp" });
  const cloudRadius = safeInterpolate(localFrame, [10, 50], [0, 140], { extrapolateRight: "clamp" });

  // Dust particles – simple circles floating outward & up
  const dustCount = 80;
  const dust = Array.from({ length: dustCount }).map((_, i) => {
    const seed = i * 137.508; // golden angle
    const angle = (seed % 360) * (Math.PI / 180);
    const speed = 1.5 + (i % 5) * 0.8;
    const dist = localFrame * speed * 0.4 + (i % 20) * 3;
    const x = 540 + Math.cos(angle) * dist;
    const y = 960 - localFrame * 0.6 * (1 + (i % 3) * 0.2) - Math.sin(angle) * dist * 0.1;
    const size = 1 + (i % 5) * 1.5;
    const opacity = Math.max(0, 1 - dist / 600 - localFrame / 200);
    const color = `hsl(30, 80%, ${50 + (i % 30)}%)`;
    return { x, y, size, opacity, color };
  });

  return (
    <AbsoluteFill
      style={{
        backgroundColor: "transparent",
        display: "flex",
        alignItems: "center",
        justifyContent: "center",
      }}
    >
      <svg
        viewBox="0 0 1080 1920"
        style={{ width: "100%", height: "100%", position: "absolute" }}
      >
        {/* Desert ground line */}
        <rect
          x={0}
          y={1100}
          width={1080}
          height={820}
          fill="rgba(160, 120, 60, 0.15)"
        />
        <line
          x1={0}
          y1={1100}
          x2={1080}
          y2={1100}
          stroke="rgba(200, 160, 60, 0.3)"
          strokeWidth={2}
        />

        {/* Background dust */}
        {dust.map((d, i) => (
          <circle
            key={i}
            cx={d.x}
            cy={d.y}
            r={d.size}
            fill={d.color}
            opacity={d.opacity * 0.5}
          />
        ))}

        {/* Shockwave rings */}
        <circle
          cx={540}
          cy={960}
          r={ring1R}
          fill="none"
          stroke="#ff6600"
          strokeWidth={4}
          opacity={ring1Opacity}
        />
        <circle
          cx={540}
          cy={960}
          r={ring2R}
          fill="none"
          stroke="#ff4400"
          strokeWidth={3}
          opacity={ring2Opacity}
        />
        <circle
          cx={540}
          cy={960}
          r={ring3R}
          fill="none"
          stroke="#ff2200"
          strokeWidth={2}
          opacity={ring3Opacity}
        />

        {/* Central flash */}
        <circle
          cx={540}
          cy={960}
          r={flashRadius}
          fill="#ffcc00"
          opacity={flashOpacity}
        />
        <circle
          cx={540}
          cy={960}
          r={flashRadius * 0.6}
          fill="#ffffff"
          opacity={flashOpacity * 0.7}
        />

        {/* Mushroom cloud – stem */}
        <path
          d={`M 460 ${960 + stemHeight * 0.1} Q 460 ${960 - stemHeight * 0.4}, 540 ${960 - stemHeight * 0.6} Q 620 ${960 - stemHeight * 0.4}, 620 ${960 + stemHeight * 0.1}`}
          fill="rgba(200, 100, 30, 0.7)"
        />
        {/* Mushroom cloud – cap */}
        <ellipse
          cx={540}
          cy={960 - stemHeight * 0.6}
          rx={cloudRadius}
          ry={cloudRadius * 0.7}
          fill="rgba(220, 120, 40, 0.8)"
        />
        <ellipse
          cx={540}
          cy={960 - stemHeight * 0.6}
          rx={cloudRadius * 0.7}
          ry={cloudRadius * 0.5}
          fill="rgba(240, 160, 60, 0.6)"
        />

        {/* Debris particles flying outward */}
        {Array.from({ length: 35 }).map((_, i) => {
          const angle = (i / 35) * Math.PI * 2;
          const speed = 4 + (i % 7) * 1.2;
          const dist = localFrame * speed * 0.5 + (i % 10) * 2;
          const x = 540 + Math.cos(angle) * dist;
          const y = 960 + Math.sin(angle) * dist * 0.5 - localFrame * 0.3;
          const size = 2 + (i % 4) * 2;
          const opacity = Math.max(0, 1 - dist / 500);
          return (
            <circle
              key={i}
              cx={x}
              cy={y}
              r={size}
              fill={i % 2 === 0 ? "#ff6600" : "#ffaa00"}
              opacity={opacity}
            />
          );
        })}

        {/* Title text with very low opacity */}
        <text
          x={540}
          y={1800}
          textAnchor="middle"
          fontSize={36}
          fontFamily="sans-serif"
          fill="rgba(255,200,100,0.4)"
          letterSpacing={4}
        >
          The Nuclear Age
        </text>
      </svg>
    </AbsoluteFill>
  );
};

export default FocusMode;
